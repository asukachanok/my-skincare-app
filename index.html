    <script src="https://cdn.tailwindcss.com"></script>

            cursor: pointer;

<body class="bg-slate-50 font-sans text-slate-800 pb-20">

 

    <div id="app" class="max-w-md mx-auto p-4">

        <!-- 標題 -->

        <header class="flex justify-between items-center mb-6">

            <div>

                <h1 class="text-3xl font-bold text-pink-600 tracking-tight">庫存管理</h1>

                <p class="text-xs text-slate-400 font-medium">資料將自動儲存於此裝置</p>

            </div>

            <button onclick="openAddModal()" class="bg-pink-500 hover:bg-pink-600 text-white p-3 rounded-full shadow-lg transition-transform active:scale-90">

                <svg xmlns="http://www.w3.org/2000/svg" width="24" height="24" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="3" stroke-linecap="round" stroke-linejoin="round"><line x1="12" y1="5" x2="12" y2="19"></line><line x1="5" y1="12" x2="19" y2="12"></line></svg>

            </button>

        </header>

 

        <!-- 分類標籤 -->

        <div class="flex gap-2 overflow-x-auto no-scrollbar mb-6" id="typeTabs">

            <button onclick="changeType('護膚')" id="tab-護膚" class="tab-btn px-6 py-2 rounded-xl text-sm font-bold bg-pink-500 text-white shadow-md">護膚</button>

            <button onclick="changeType('化妝')" id="tab-化妝" class="tab-btn px-6 py-2 rounded-xl text-sm font-bold text-slate-400">化妝</button>

            <button onclick="changeType('日常')" id="tab-日常" class="tab-btn px-6 py-2 rounded-xl text-sm font-bold text-slate-400">日常</button>

        </div>

 

        <!-- 產品列表區 -->

        <div id="productList" class="space-y-6">

            <!-- 內容由 JS 生成 -->

        </div>

    </div>

 

    <!-- 新增產品彈窗 -->

    <div id="addModal" class="fixed inset-0 bg-slate-900/60 backdrop-blur-sm hidden flex items-center justify-center p-4 z-50">

        <div class="bg-white rounded-[2rem] w-full max-w-sm p-8 shadow-2xl">

            <h2 class="text-xl font-bold mb-6 text-slate-800">新增產品</h2>

            <div class="space-y-4">

                <div>

                    <label class="text-[10px] font-black text-slate-400 uppercase ml-1">品牌</label>

                    <input id="newBrand" type="text" placeholder="例如：Fiole" class="w-full bg-slate-50 p-4 rounded-xl text-sm outline-none focus:ring-2 ring-pink-100">

                </div>

                <div>

                    <label class="text-[10px] font-black text-slate-400 uppercase ml-1">產品名稱</label>

                    <input id="newName" type="text" placeholder="例如：補色洗頭水" class="w-full bg-slate-50 p-4 rounded-xl text-sm outline-none focus:ring-2 ring-pink-100">

                </div>

                <div>

                    <label class="text-[10px] font-black text-slate-400 uppercase ml-1">類別</label>

                    <select id="newCategory" class="w-full bg-slate-50 p-4 rounded-xl text-sm outline-none">

                        <!-- 類別會根據目前的 Type 動態生成 -->

                    </select>

                </div>

                <div class="pt-2">

                    <button onclick="saveProduct()" class="w-full py-4 bg-pink-500 hover:bg-pink-600 text-white rounded-2xl font-bold shadow-lg transition-all active:scale-95">確認新增</button>

                    <button onclick="closeAddModal()" class="w-full mt-2 py-2 text-slate-400 text-sm font-medium">取消</button>

                </div>

            </div>

        </div>

    </div>

 

    <script>

        // --- 初始化資料 ---

        let currentType = '護膚';

        let items = JSON.parse(localStorage.getItem('glow_tracker_data')) || [

            {

                id: Date.now(),

                brand: 'Fiole',

                name: 'Purple Shampoo',

                type: '日常',

                category: '補色洗頭水',

                batches: [

                    { id: 'b1', status: '使用中', sizeType: '正裝', quantity: 1, price: '90', purchaseDate: '2025-01-05', location: 'United Hair', openDate: '2025-01-06', finishDate: '' }

                ]

            }

        ];

 

        const categoriesMap = {

            '護膚': ['洗面', '化妝水', '精華', '面霜', '卸妝'],

            '化妝': ['防曬', '底妝', '眼影', '唇膏'],

            '日常': ['洗頭水', '補色洗頭水', '護髮素', '沐浴露']

        };

 

        // --- 核心功能 ---

        function saveToLocal() {

            localStorage.setItem('glow_tracker_data', JSON.stringify(items));

            render();

        }

 

        function changeType(type) {

            currentType = type;

            document.querySelectorAll('.tab-btn').forEach(btn => {

                btn.classList.remove('bg-pink-500', 'text-white', 'shadow-md');

                btn.classList.add('text-slate-400');

            });

            const active = document.getElementById(`tab-${type}`);

            active.classList.add('bg-pink-500', 'text-white', 'shadow-md');

            active.classList.remove('text-slate-400');

            render();

        }

 

        function openAddModal() {

            const select = document.getElementById('newCategory');

            select.innerHTML = categoriesMap[currentType].map(c => `<option value="${c}">${c}</option>`).join('');

            document.getElementById('addModal').classList.remove('hidden');

        }

 

        function closeAddModal() {

            document.getElementById('addModal').classList.add('hidden');

        }

 

        function saveProduct() {

            const brand = document.getElementById('newBrand').value;

            const name = document.getElementById('newName').value;

            const category = document.getElementById('newCategory').value;

            if(!name) return alert('請輸入產品名稱');

 

            items.push({

                id: Date.now(),

                brand,

                name,

                type: currentType,

                category,

                batches: []

            });

            saveToLocal();

            closeAddModal();

        }

 

        function addBatch(productId) {

            const product = items.find(p => p.id === productId);

            product.batches.push({

                id: 'b' + Date.now(),

                status: '未開封',

                sizeType: '正裝',

                quantity: 1,

                price: '',

                purchaseDate: new Date().toISOString().split('T')[0],

                location: '',

                openDate: '',

                finishDate: ''

            });

            saveToLocal();

        }

 

        function updateBatch(productId, batchId, field, value) {

            const product = items.find(p => p.id === productId);

            const batch = product.batches.find(b => b.id === batchId);

            batch[field] = value;

            saveToLocal();

        }

 

        function cycleStatus(productId, batchId) {

            const product = items.find(p => p.id === productId);

            const batch = product.batches.find(b => b.id === batchId);

            const states = ['未開封', '使用中', '已用完'];

            let idx = (states.indexOf(batch.status) + 1) % 3;

            batch.status = states[idx];

           

            if (batch.status === '使用中' && !batch.openDate) {

                batch.openDate = new Date().toISOString().split('T')[0];

            }

            if (batch.status === '已用完' && !batch.finishDate) {

                batch.finishDate = new Date().toISOString().split('T')[0];

            }

            saveToLocal();

        }

 

        function deleteProduct(id) {

            if(confirm('確定要刪除此產品及其所有批次嗎？')) {

                items = items.filter(i => i.id !== id);

                saveToLocal();

            }

        }

 

        // --- 渲染 UI ---

        function render() {

            const list = document.getElementById('productList');

            const filtered = items.filter(i => i.type === currentType);

 

            if (filtered.length === 0) {

                list.innerHTML = `<div class="text-center py-20 text-slate-300 font-bold">目前沒有產品</div>`;

                return;

            }

 

            list.innerHTML = filtered.map(product => {

                const inUse = product.batches.filter(b => b.status === '使用中').reduce((s, b) => s + (parseInt(b.quantity) || 0), 0);

                const stock = product.batches.filter(b => b.status === '未開封').reduce((s, b) => s + (parseInt(b.quantity) || 0), 0);

 

                return `

                <div class="bg-white rounded-[2rem] shadow-sm border border-slate-100 overflow-hidden transition-all">

                    <!-- 產品頭部 -->

                    <div class="p-5 flex justify-between items-start">

                        <div>

                            <div class="text-[10px] font-black text-pink-500 uppercase tracking-widest mb-1">${product.brand || '未知品牌'}</div>

                            <div class="flex items-center gap-2 flex-wrap">

                                <h3 class="text-lg font-black text-slate-800">${product.name}</h3>

                                <div class="flex gap-1">

                                    ${inUse > 0 ? `<span class="bg-green-100 text-green-600 text-[9px] font-black px-2 py-0.5 rounded-full">${inUse} 用</span>` : ''}

                                    ${stock > 0 ? `<span class="bg-blue-100 text-blue-600 text-[9px] font-black px-2 py-0.5 rounded-full">${stock} 存</span>` : ''}

                                </div>

                            </div>

                            <div class="text-[10px] text-slate-400 font-bold mt-1"># ${product.category}</div>

                        </div>

                        <button onclick="deleteProduct(${product.id})" class="text-slate-200 hover:text-red-400 p-2">

                            <svg xmlns="http://www.w3.org/2000/svg" width="18" height="18" viewBox="0 0 24 24" fill="none" stroke="currentColor" stroke-width="2" stroke-linecap="round" stroke-linejoin="round"><polyline points="3 6 5 6 21 6"></polyline><path d="M19 6v14a2 2 0 0 1-2 2H7a2 2 0 0 1-2-2V6m3 0V4a2 2 0 0 1 2-2h4a2 2 0 0 1 2 2v2"></path></svg>

                        </button>

                    </div>

 

                    <!-- 批次列表 -->

                    <div class="bg-slate-50/50 p-4 border-t border-slate-100">

                        <div class="flex justify-between items-center mb-3 px-1">

                            <span class="text-[9px] font-black text-slate-400 uppercase tracking-widest">購入批次詳情</span>

                            <button onclick="addBatch(${product.id})" class="text-pink-500 text-[9px] font-black hover:underline">+ 新增購入</button>

                        </div>

                        <div class="space-y-3">

                            ${product.batches.map(batch => `

                                <div class="bg-white p-3 rounded-2xl border border-slate-200 shadow-sm ${batch.status === '已用完' ? 'opacity-40 grayscale' : ''}">

                                    <!-- 第一行 -->

                                    <div class="grid grid-cols-[70px_1fr] gap-3 items-center mb-2 pb-2 border-b border-slate-50">

                                        <button onclick="cycleStatus(${product.id}, '${batch.id}')"

                                            class="py-1 rounded-lg text-[8px] font-black uppercase tracking-tighter ${batch.status === '使用中' ? 'bg-green-500 text-white' : (batch.status === '已用完' ? 'bg-slate-300 text-white' : 'bg-slate-100 text-slate-400')}">

                                            ${batch.status}

                                        </button>

                                        <div class="grid grid-cols-[80px_60px_1fr] gap-2 text-[10px]">

                                            <div class="flex items-center gap-1 font-black text-slate-600 border-r border-slate-100 pr-1">

                                                <span class="text-slate-300 shrink-0">規格</span>

                                                <input type="text" value="${batch.sizeType}" onchange="updateBatch(${product.id}, '${batch.id}', 'sizeType', this.value)" class="w-full bg-transparent outline-none">

                                            </div>

                                            <div class="flex items-center gap-1 font-black text-slate-600 border-r border-slate-100 px-1">

                                                <span class="text-slate-300 shrink-0">數量</span>

                                                <input type="number" value="${batch.quantity}" onchange="updateBatch(${product.id}, '${batch.id}', 'quantity', this.value)" class="w-full bg-transparent outline-none">

                                            </div>

                                            <div class="flex items-center gap-1 font-mono font-black text-blue-600">

                                                <span class="text-slate-300">$</span>

                                                <input type="text" value="${batch.price}" onchange="updateBatch(${product.id}, '${batch.id}', 'price', this.value)" class="w-full bg-transparent outline-none" placeholder="價格">

                                            </div>

                                        </div>

                                    </div>

                                    <!-- 第二行 -->

                                    <div class="grid grid-cols-2 gap-y-2 gap-x-4 text-[9px]">

                                        <div class="flex flex-col">

                                            <span class="text-[8px] font-black text-slate-300 uppercase">購入日期</span>

                                            <input type="date" value="${batch.purchaseDate}" onchange="updateBatch(${product.id}, '${batch.id}', 'purchaseDate', this.value)" class="font-bold text-slate-500 outline-none bg-transparent">

                                        </div>

                                        <div class="flex flex-col">

                                            <span class="text-[8px] font-black text-slate-300 uppercase">購買地點</span>

                                            <input type="text" value="${batch.location}" onchange="updateBatch(${product.id}, '${batch.id}', 'location', this.value)" class="font-bold text-slate-500 outline-none bg-transparent" placeholder="地點">

                                        </div>

                                        <div class="flex flex-col">

                                            <span class="text-[8px] font-black text-pink-300 uppercase">開封日期</span>

                                            <input type="date" value="${batch.openDate}" onchange="updateBatch(${product.id}, '${batch.id}', 'openDate', this.value)" class="font-black text-pink-500 outline-none bg-transparent">

                                        </div>

                                        <div class="flex flex-col">

                                            <span class="text-[8px] font-black text-slate-300 uppercase">用完日期</span>

                                            <input type="date" value="${batch.finishDate}" onchange="updateBatch(${product.id}, '${batch.id}', 'finishDate', this.value)" class="font-black text-slate-500 outline-none bg-transparent">

                                        </div>

                                    </div>

                                </div>

                            `).join('')}

                        </div>

                    </div>

                </div>

                `;

            }).join('');

        }

 

        // 初始化渲染

        render();

    </script>

</body>
